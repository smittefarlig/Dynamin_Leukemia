run("8-bit");

//Generate timepoint images

rename("red.tif");

//Binarisation

//red
setAutoThreshold("Intermodes dark");

run("Convert to Mask");

//Insert a path manually

run("Save", "save=//.../red.tif");

//Switch to green

close();
close();

//run("Threshold...");

//Insert a path manually
